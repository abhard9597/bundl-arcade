188 inline void DrawScaledBackground(
189 wxDC &dc,
190 const wxBitmap &backgroundBitmap,
191 const wxSize &size,
192 bool darkMode,
193 double zoom = 1.0)
194 {
195 if (backgroundBitmap.IsOk())
196 {
197 wxImage bg = backgroundBitmap.ConvertToImage();
198
199 double imgW = static_cast<double>(bg.GetWidth());
200 double imgH = static_cast<double>(bg.GetHeight());
201 double winW = static_cast<double>(size.GetWidth());
202 double winH = static_cast<double>(size.GetHeight());
203
204 double baseScale = std::max(winW / imgW, winH / imgH);
205 double finalScale = baseScale * zoom;
206
207 int drawW = static_cast<int>(imgW * finalScale);
208 int drawH = static_cast<int>(imgH * finalScale);
209
210 bg.Rescale(drawW, drawH, wxIMAGE_QUALITY_HIGH);
211
212 int x = (size.GetWidth() - drawW) / 2;
213 int y = (size.GetHeight() - drawH) / 2;
214
215 dc.DrawBitmap(wxBitmap(bg), x, y, false);
216 }
217 else
218 {
219 dc.SetBrush(wxBrush(FrameBackground(darkMode)));
220 dc.SetPen(*wxTRANSPARENT_PEN);
221 dc.DrawRectangle(0, 0, size.GetWidth(), size.GetHeight());
222 }
223
224 dc.SetBrush(wxBrush(OverlayColour(darkMode)));
225 dc.SetPen(*wxTRANSPARENT_PEN);
226 dc.DrawRectangle(0, 0, size.GetWidth(), size.GetHeight());
227 }
